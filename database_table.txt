create database acmDB;

CREATE TABLE person (
    username VARCHAR(255) NOT NULL,
    PRIMARY KEY(username), 
    password VARCHAR(255) NOT NULL,
    usertype VARCHAR(30) NOT NULL,
    email VARCHAR(255) NOT NULL,
    score INT,
    teamname VARCHAR(255),
    lastdate DATE,
    name VARCHAR(30),
    gender VARCHAR(30),
    Tshirtsize VARCHAR(30),
    studyyear INT,
    major VARCHAR(255),
    tel VARCHAR(255)
);
